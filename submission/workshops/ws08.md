```python
@app.route('/dict/<string:word>')
def my_dict(word):
    d = {
        'apple': '사과',
        'banana': '바나나',
    }
    if not d.get(word):
        return f'{word}은(는) 나만의 단어장에 없는 단어입니다!'
    return f'{word}은(는) {d[word]}!'

if __name__ == '__main__':
    app.run(debug=True)
```
